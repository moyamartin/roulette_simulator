33 void RouletteBets::setResult(int random_number)
34 {
35  // Stores the random value generated by Roulette
36  simple_bets_result[VALUE] = random_number;
37  // Roulette first resets the map calling resetResult
38  /* if random_number is zero then it does not updates the values
39  * indicating that everyone has lost
40  */
41  if(random_number > 0) {
42  // First check if the number is even or not and updates the EVEN and ODD
43  // values in simple_bet_result
44  simple_bets_result[EVEN] = (random_number%2 == 0);
45  simple_bets_result[ODD] = !simple_bets_result[EVEN];
46 
47  // The numbers on the roulette are separated by HIGH or LOW if it is
48  // bigger or not than 19
49  simple_bets_result[LOW] = (random_number < 19);
50  simple_bets_result[HIGH] = !simple_bets_result[LOW];
51 
52  /*
53  * If the number is LOW or HIGH and less or equal than 10 or 28
54  * (respectively) then the EVEN numbers are BLACK and the ODD numbers
55  * are RED
56  * If the numbers is LOW or HIGH and higher than 10 or 28 (respectively)
57  * then the ODD numbers are BLACK and the EVEN numbers are RED
58  * Please check the color scheme of the table on the README for a better
59  * explanation
60  */
61  if((simple_bets_result[LOW] && random_number <= 10) ||
62  (simple_bets_result[HIGH] && random_number <= 28)){
63  simple_bets_result[BLACK] = simple_bets_result[EVEN];
64  } else {
65  simple_bets_result[BLACK] = simple_bets_result[ODD];
66  }
67  simple_bets_result[RED] = !simple_bets_result[BLACK];
68  }
69 }
